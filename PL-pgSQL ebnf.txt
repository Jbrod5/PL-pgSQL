NOTACION EBNF:
| -> or
[] -> opcional (cero o una vez)
{} -> repeticion (cero o mÃ¡s veces)
() -> agrupacion 
"..."_"..." -> rango de caracteres (como [a-z])
; -> final de regla



//- - - - - - - - - - - - - -  Cosas generales - - - - - - - - - - - - - - 

letra = ("a"_"z"|"A"_"Z");
digito = "0"_"9";
identificador = (letra | "_") {letra | digito | "_"};
puntocoma = ";";
asignacion = ":=";



// - - - - - - - - - - - - Declaraciones y tipado - - - - - - - - - - - -

char = "CHAR";
boolean = "BOOLEAN";
